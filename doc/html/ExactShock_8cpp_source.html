00193 void SHOCK::STARTE(REAL& p)
00194 {
00195   REAL TOL=1.0e-6;
00196   INTEGER QMAX=2.0;
00197 
00198   //compute guess value from pvrs
00199   REAL pV=0.5*(pL+pR)-0.125*(uR-uL)*(rhoL+rhoR)*(cL+cR);
00200   REAL pMin=min(pL,pR);
00201   REAL pMax=max(pL,pR);
00202   REAL QRAT=pMax/pMin;
00203 
00204   if(QRAT<=QMAX&&(pMin<=pV&&pV<=pMax)){
00205     //use pvrs solution as guess
00206     p=max(TOL,pV);
00207   }
00208   else{
00209     if(pV<pMin){
00210       //use two-rarefaction solution
00211       REAL pNU=cL+cR-G7*(uR-uL);
00212       REAL pDE=cL/pow(pL,G1)+cR/pow(pR,G1);
00213       p=pow((pNU/pDE),G3);
00214     }
00215     else{
00216       //two-shock approximation with pv as estimate
00217       REAL GEL=sqrt((G5/rhoL)/(G6*pL+max(TOL,pV)));
00218       REAL GER=sqrt((G5/rhoR)/(G6*pR+max(TOL,pV)));
00219       p=(GEL*pL+GER*pR-(uR-uL))/(GEL+GER);
00220       p=max(TOL,p);
00221     }
00222   }
