00950 void NumericalSHOCK::FaceFluxHLLCBACKUP(REAL rhoL, REAL uL, REAL pL,

00952                                   REAL& MF1, REAL& MF2, REAL& MF3)
00953 {
00954   //0->basic variables for left and right state
00955   REAL cL=sqrt(GAMMA*pL/rhoL);  
00956   REAL HL=GAMMA/(GAMMA-1)*pL/rhoL+0.5*uL*uL; 
00958   REAL cR=sqrt(GAMMA*pR/rhoR);   
00959   REAL HR=GAMMA/(GAMMA-1)*pR/rhoR+0.5*uR*uR; 
00961   //1->Roe's average
00962   REAL f=sqrt(rhoR/rhoL);
00963   REAL Au=(uL+uR*f)/(1+f);
00964   REAL AH=(HL+HR*f)/(1+f);
00965   REAL Ac=sqrt((GAMMA-1)*(AH-0.5*uL*uL));
00966 
00967   //2->Normal velocity
00968   REAL AU=Au;
00969 
00970   //3->Left state wave speed
00971   REAL UL=uL;
00972   REAL SL=min(UL-cL,AU-Ac);
00973 
00974   //4->Right state wave speed
00975   REAL UR=uR;
00976   REAL SR=max(UR+cR,AU+Ac);
00977 
00978   //5->Middle state wave speed
00979   REAL g1=rhoR*UR*(SR-UR)-rhoL*UL*(SL-UL)+(pL-pR);
00980   REAL g2=rhoR*(SR-UR)-rhoL*(SL-UL);
00981   REAL SM=g1/g2;
00982 
00983   //6->The interface flux
00984   if(SL>0){
00985     //F=FL
00986     MF1=rhoL*UL;
00987     MF2=rhoL*uL*UL+pL;
00988     MF3=rhoL*HL*UL;    
00989     return;
00990   }
00991 
00992   if(SL<=0&&0<SM){
00993     //F=FL_star
00994     REAL w1L=rhoL;              
00995     REAL w2L=rhoL*uL;           
00996     REAL w3L=pL/(GAMMA-1)+0.5*rhoL*uL*uL; 
00998     REAL F1L=rhoL*UL;           
00999     REAL F2L=rhoL*uL*UL+pL;     
01000     REAL F3L=rhoL*HL*UL;        
01002     REAL pM=rhoL*(UL-SL)*(UL-SM)+pL; 
01004     REAL f1=SL-UL;
01005     REAL f2=SL-SM;
01006     REAL f3=f1/f2;
01007     REAL f4=(pM-pL);
01008     REAL f5=f4/f2;
01009 
01010     REAL w1M=rhoL*f3;           
01011     REAL w2M=rhoL*uL*f3+f5;     
01012     REAL w3M=w3L*f3+(-pL*UL+pM*SM)/f2; 
01014     MF1=F1L+SL*(w1M-w1L);
01015     MF2=F2L+SL*(w2M-w2L);
01016     MF3=F3L+SL*(w3M-w3L);
01017 
01018     return;
01019   }
01020 
01021   if(SM<=0&&0<=SR){
01022     //F=FR_star
01023     REAL w1R=rhoR;              
01024     REAL w2R=rhoR*uR;           
01025     REAL w3R=pR/(GAMMA-1)+0.5*rhoR*uR*uR; 
01027     REAL F1R=rhoR*UR;           
01028     REAL F2R=rhoR*uR*UR+pR;     
01029     REAL F3R=rhoR*HR*UR;        
01031     REAL pM=rhoL*(UL-SL)*(UL-SM)+pL; 
01033     REAL f1=SR-UR;
01034     REAL f2=SR-SM;

01036     REAL f4=(pM-pR);
01037     REAL f5=f4/f2;
01038 
01039     REAL w1M=rhoR*f3;           
01040     REAL w2M=rhoR*uR*f3+f5;     
01041     REAL w3M=w3R*f3+(-pR*UR+pM*SM)/f2; 
01043     MF1=F1R+SR*(w1M-w1R);
01044     MF2=F2R+SR*(w2M-w2R);
01045     MF3=F3R+SR*(w3M-w3R);
01046 
01047     return;
01048   }
01049 
01050   if(SR<0){
01051     //F=FR
01052     MF1=rhoR*UR;
01053     MF2=rhoR*uR*UR+pR;
01054     MF3=rhoR*HR*UR;
01055     return;
